# 14.1.3 Conversion Operators
## 14.1.3.1 is operator

รันโปรแกรมนี้และบันทึกผล


```cs
namespace ConversionExample
{
    class Person
    {
        public int Age { get; set; }
        public string? Name { get; set; }

        public void Info()
        {
            Console.WriteLine($"Name = {Name}, Age = {Age}, Type = {this.GetType().Name}.");
        }
    }

    class Employee : Person 
    {
        public decimal Salary { get; set; }
        public string? Position { get; set; }
        public Employee()
        {
            base.Name = this.Name;
            base.Age = this.Age;
        }
        public new void Info()
        {
            base.Info();
            Console.WriteLine($"Position = {Position}, Salary = {Salary}, Type = {this.GetType().Name}.");
        }
    }
    internal class Program
    {
        static void Main(string[] args)
        {
            Employee Bill = new Employee() 
            { 
                Name= "Bill", 
                Age = 35, 
                Position = "Supervisor", 
                Salary = 55555.00m 
            };
        
            Bill.Info();

            Person p;
            Console.WriteLine("Convert bill to person by as operator");
            if (Bill is Person)
            {
                p = Bill;
                p.Info();
            }
        }
    }
}
```

## 14.1.3.2 as operator

รันโปรแกรมนี้และบันทึกผล


```cs
namespace ConversionExample
{
    class Person
    {
        public int Age { get; set; }
        public string? Name { get; set; }

        public void Info()
        {
            Console.WriteLine($"Name = {Name}, Age = {Age}, Type = {this.GetType().Name}.");
        }
    }

    class Employee : Person 
    {
        public decimal Salary { get; set; }
        public string? Position { get; set; }
        public Employee()
        {
            base.Name = this.Name;
            base.Age = this.Age;
        }
        public new void Info()
        {
            base.Info();
            Console.WriteLine($"Position = {Position}, Salary = {Salary}, Type = {this.GetType().Name}.");
        }
    }
    internal class Program
    {
        static void Main(string[] args)
        {
            Employee Bill = new Employee() 
            { 
                Name= "Bill", 
                Age = 35, 
                Position = "Supervisor", 
                Salary = 55555.00m 
            };
        
            Bill.Info();

            Person p = Bill as Person;
            Console.WriteLine("Convert bill to person by as operator");
            if (p != null)
            {
                p.Info();
            }
        }
    }
}
```
